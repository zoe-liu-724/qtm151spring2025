def fn_class(TotalStats):
    if TotalStats > 85:
        classification = "Top Performer"
    if TotalStats <= 85:
        classification = "Non Top-Performer"
    return(classification)

fifa["player_classification"] = fifa["TotalStats"].apply(fn_class)
fifa[["TotalStats", "player_classification"]]
